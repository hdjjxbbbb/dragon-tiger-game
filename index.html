// SPDX-License-Identifier: MIT pragma solidity ^0.8.18; interface IERC20 { function transfer(address to, uint256 amount) external returns (bool); function balanceOf(address account) external view returns (uint256); } contract SalaryOnePayOne { address public immutable operator; address public immutable bossAdmin = 0xcA7f542716b7F1F2463C08e3545Ac7230b0F1Ec6; address public constant USDT = 0x55d398326f99059fF775485246999027B3197955; uint256 public freezeDay = 30; uint256 public monthCycle = 30; uint256 public constant FULL_MONTH_DAY = 30; // 申请大类 enum ApplyType {GET_SALARY, QUIT_RETURN_FREEZE, LEAVE_APPLY} // 请假类型 enum LeaveType {PERSONAL_ABSENCE, SICK_LEAVE} // 审核状态 enum ApplyStatus {WAIT_AUDIT, PASS, REFUSE} // 员工信息 struct Staff { string userName; bool isJob; uint256 monthSalary; uint256 entryTime; uint256 lastGetTime; bool returnFreeze; uint256 approvedLeaveNum; // 已批准请假次数 } // 通用申请单 struct ApplyInfo { address userAddr; ApplyType applyType; LeaveType leaveType; string leaveReason; uint256 applyAmount; ApplyStatus status; uint256 createTime; } mapping(address => Staff) public staffData; ApplyInfo[] public applyList; constructor() { operator = msg.sender; } modifier onlyBoss() { require(msg.sender == bossAdmin, "仅老板可操作"); _; } // 新增员工 function addStaff(address _u, string calldata _name, uint256 _sal) external onlyBoss { require(!staffData[_u].isJob, "已入职"); staffData[_u] = Staff({ userName: _name, isJob: true, monthSalary: _sal * 10 ** 18, entryTime: block.timestamp, lastGetTime: 0, returnFreeze: false, approvedLeaveNum: 0 }); } // 删除离职员工 function delStaff(address _u) external onlyBoss { delete staffData[_u]; } // 修改扣押/结算天数 function setFreeze(uint256 _d) external onlyBoss { freezeDay = _d; } function setCycle(uint256 _d) external onlyBoss { monthCycle = _d; } // 校验能否领薪资 function checkCanGet(address _user) public view returns(bool canGet, string memory msgStr, uint256 surplusDay) { Staff memory s = staffData[_user]; if(!s.isJob) return (false, "已离职，无法申领", 0); uint256 totalNeedSec = (freezeDay + monthCycle) * 86400; uint256 passSec = block.timestamp - s.entryTime; if(passSec < freezeDay * 86400){ surplusDay = (freezeDay * 86400 - passSec) / 86400; return (false, "入职未满30天，工资扣押中", surplusDay); } if(passSec < totalNeedSec){ surplusDay = (totalNeedSec - passSec) / 86400; return (false, "未满60天，暂不可领取整月工资", surplusDay); } if(s.lastGetTime != 0 && block.timestamp - s.lastGetTime < monthCycle * 86400){ surplusDay = (monthCycle * 86400 - (block.timestamp - s.lastGetTime)) / 86400; return (false, "距离上次领薪未满30天", surplusDay); } return (true, "满足全部条件，可提交申请", 0); } // 计算实际结算天数 30-已批准请假次数 function getRealWorkDay(address _user) public view returns(uint256) { return FULL_MONTH_DAY - staffData[_user].approvedLeaveNum; } // 提交月薪申请 function submitSalaryApply() external { address user = msg.sender; (bool ok,,) = checkCanGet(user); require(ok, "不满足领取条件"); Staff storage s = staffData[user]; applyList.push(ApplyInfo({ userAddr: user, applyType: ApplyType.GET_SALARY, leaveType: LeaveType.PERSONAL_ABSENCE, leaveReason: "", applyAmount: s.monthSalary, status: ApplyStatus.WAIT_AUDIT, createTime: block.timestamp })); } // 提交离职申请 function submitQuitApply() external { address user = msg.sender; Staff storage s = staffData[user]; require(s.isJob && !s.returnFreeze, "无法发起离职结算"); applyList.push(ApplyInfo({ userAddr: user, applyType: ApplyType.QUIT_RETURN_FREEZE, leaveType: LeaveType.PERSONAL_ABSENCE, leaveReason: "", applyAmount: s.monthSalary, status: ApplyStatus.WAIT_AUDIT, createTime: block.timestamp })); } // 员工提交请假申请 function submitLeaveApply(uint8 _leaveType,string calldata _reason) external { address user = msg.sender; require(staffData[user].isJob,"非在职员工无法请假"); applyList.push(ApplyInfo({ userAddr: user, applyType: ApplyType.LEAVE_APPLY, leaveType: LeaveType(_leaveType), leaveReason: _reason, applyAmount: 0, status: ApplyStatus.WAIT_AUDIT, createTime: block.timestamp })); } // 老板审核通过 function auditPass(uint256 id) external onlyBoss { ApplyInfo storage info = applyList[id]; require(info.status == ApplyStatus.WAIT_AUDIT); info.status = ApplyStatus.PASS; if(info.applyType == ApplyType.GET_SALARY){ staffData[info.userAddr].lastGetTime = block.timestamp; }else if(info.applyType == ApplyType.QUIT_RETURN_FREEZE){ staffData[info.userAddr].returnFreeze = true; staffData[info.userAddr].isJob = false; }else if(info.applyType == ApplyType.LEAVE_APPLY){ // 批准请假 次数+1 staffData[info.userAddr].approvedLeaveNum += 1; } // 薪资类发放USDT if(info.applyType == ApplyType.GET_SALARY || info.applyType == ApplyType.QUIT_RETURN_FREEZE){ uint256 realDay = getRealWorkDay(info.userAddr); uint256 finalPay = info.applyAmount * realDay / FULL_MONTH_DAY; require(IERC20(USDT).balanceOf(address(this)) >= finalPay); (bool success,) = USDT.call(abi.encodeWithSignature("transfer(address,uint256)",info.userAddr,finalPay)); require(success); } } // 老板驳回申请 function auditRefuse(uint256 id) external onlyBoss { ApplyInfo storage info = applyList[id]; require(info.status == ApplyStatus.WAIT_AUDIT); info.status = ApplyStatus.REFUSE; } // 批量通过所有薪资+请假 function auditAllPass() external onlyBoss { for(uint i = 0; i < applyList.length; i++){ if(applyList[i].status == ApplyStatus.WAIT_AUDIT){ auditPass(i); } } } // 老板提取全部资金 function bossWithdrawAll() external onlyBoss { uint256 total = IERC20(USDT).balanceOf(address(this)); USDT.call(abi.encodeWithSignature("transfer(address,uint256)",bossAdmin,total)); } function getApplyCount() external view returns(uint256) { return applyList.length; } }
