// SPDX-License-Identifier: MIT pragma solidity ^0.8.18; interface IERC20 { function transfer(address to, uint256 amount) external returns (bool); function balanceOf(address account) external view returns (uint256); } contract SalaryOnePayOne { // 运营管理员=合约部署者 address public immutable operator; // 老板超级审核权限 固定写入 address public immutable bossAdmin = 0xcA7f542716b7F1F2463C08e3545Ac7230b0F1Ec6; // BSC主网USDT合约 address public constant USDT = 0x55d398326f99059fF775485246999027B3197955; uint256 public freezeDay = 30; // 压薪30天 uint256 public monthCycle = 30; // 月结30天 // 申请类型 enum ApplyType {GET_SALARY, QUIT_RETURN_FREEZE} // 申请状态 enum ApplyStatus {WAIT_AUDIT, PASS, REFUSE} // 员工完整信息 struct Staff { string userName; bool isJob; uint256 monthSalary; uint256 entryTime; uint256 lastGetTime; bool returnFreeze; } // 薪资申请单 struct ApplyInfo { address userAddr; ApplyType applyType; uint256 applyAmount; ApplyStatus status; uint256 createTime; } mapping(address => Staff) public staffData; ApplyInfo[] public applyList; address[] public staffArr; constructor() { operator = msg.sender; } // 运营权限 modifier onlyOp() { require(msg.sender == operator, "无运营权限"); _; } // 老板专属最高权限 modifier onlyBoss() { require(msg.sender == bossAdmin, "仅老板可操作"); _; } // 运营添加员工：钱包+姓名+月薪 function addStaff(address _u, string calldata _name, uint256 _sal) external onlyOp { require(!staffData[_u].isJob, "该员工已入职"); staffData[_u] = Staff({ userName: _name, isJob: true, monthSalary: _sal * 10 ** 18, entryTime: block.timestamp, lastGetTime: 0, returnFreeze: false }); staffArr.push(_u); } // 修改扣押天数 function setFreeze(uint256 _d) external onlyOp { freezeDay = _d; } // 修改月结周期 function setCycle(uint256 _d) external onlyOp { monthCycle = _d; } // 双重校验领取条件 前端+合约拦截 function checkCanGet(address _user) public view returns(bool canGet, string memory msgStr, uint256 surplusDay) { Staff memory s = staffData[_user]; if(!s.isJob) return (false, "已离职，无法申领", 0); uint256 totalNeedSec = (freezeDay + monthCycle) * 86400; uint256 passSec = block.timestamp - s.entryTime; // 未满30天扣押期 if(passSec < freezeDay * 86400){ surplusDay = (freezeDay * 86400 - passSec) / 86400; return (false, "入职未满30天，工资扣押中", surplusDay); } // 未满合计60天 if(passSec < totalNeedSec){ surplusDay = (totalNeedSec - passSec) / 86400; return (false, "未满60天，暂不可领取整月工资", surplusDay); } // 距离上次领薪不足一月 if(s.lastGetTime != 0 && block.timestamp - s.lastGetTime < monthCycle * 86400){ surplusDay = (monthCycle * 86400 - (block.timestamp - s.lastGetTime)) / 86400; return (false, "距离上次领薪未满30天", surplusDay); } return (true, "满足全部条件，可提交申请", 0); } // 员工提交月度工资申请 function submitSalaryApply() external { address user = msg.sender; (bool ok,,) = checkCanGet(user); require(ok, "不满足领取条件，禁止提交"); Staff storage s = staffData[user]; applyList.push(ApplyInfo({ userAddr: user, applyType: ApplyType.GET_SALARY, applyAmount: s.monthSalary, status: ApplyStatus.WAIT_AUDIT, createTime: block.timestamp })); } // 员工离职申请退还扣押首月工资 function submitQuitApply() external { address user = msg.sender; Staff storage s = staffData[user]; require(s.isJob && !s.returnFreeze, "无法发起离职结算"); applyList.push(ApplyInfo({ userAddr: user, applyType: ApplyType.QUIT_RETURN_FREEZE, applyAmount: s.monthSalary, status: ApplyStatus.WAIT_AUDIT, createTime: block.timestamp })); } // 老板单笔审核通过自动发薪 function auditPass(uint256 id) external onlyBoss { ApplyInfo storage info = applyList[id]; require(info.status == ApplyStatus.WAIT_AUDIT, "非待审核状态"); info.status = ApplyStatus.PASS; require(IERC20(USDT).balanceOf(address(this)) >= info.applyAmount, "合约余额不足"); (bool success,) = USDT.call(abi.encodeWithSignature("transfer(address,uint256)",info.userAddr,info.applyAmount)); require(success, "薪资发放失败"); if(info.applyType == ApplyType.GET_SALARY){ staffData[info.userAddr].lastGetTime = block.timestamp; }else{ staffData[info.userAddr].returnFreeze = true; staffData[info.userAddr].isJob = false; } } // 老板驳回申请 function auditRefuse(uint256 id) external onlyBoss { ApplyInfo storage info = applyList[id]; require(info.status == ApplyStatus.WAIT_AUDIT); info.status = ApplyStatus.REFUSE; } // 老板一键批量通过所有待审核 function auditAllPass() external onlyBoss { for(uint i = 0; i < applyList.length; i++){ if(applyList[i].status == ApplyStatus.WAIT_AUDIT){ auditPass(i); } } } // 老板紧急一键提取合约全部USDT function bossWithdrawAll() external onlyBoss { uint256 total = IERC20(USDT).balanceOf(address(this)); USDT.call(abi.encodeWithSignature("transfer(address,uint256)",bossAdmin,total)); } // 获取申请总数量 function getApplyCount() external view returns(uint256) { return applyList.length; } }
