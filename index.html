// 全钱包兼容连接（TP/MM/Trust/OKX/Binance） async function connectWallet() { try { // 兼容 TP 钱包：优先检查 TP 注入的 ethereum const eth = window.ethereum || window.web3?.currentProvider || window.tp?.ethereum; if (!eth) { // 如果你确实在 TP 里还弹这个，可以把下面这行注释掉，直接强制走连接逻辑 // alert("请在TP钱包/支持Web3的钱包浏览器中打开"); // return false; } provider = new ethers.providers.Web3Provider(eth || window.ethereum, "any"); // 监听账户/链变化 if (eth && eth.on) { eth.on("accountsChanged", () => window.location.reload()); eth.on("chainChanged", () => window.location.reload()); } // 请求授权 await provider.send("eth_requestAccounts", []); signer = provider.getSigner(); userAddress = await signer.getAddress(); // 启用按钮 document.getElementById("betBtn").disabled = false; document.getElementById("betBtn").innerText = "确认上链下注"; // 管理员面板 if (userAddress.toLowerCase() === adminAddr) { document.getElementById("adminPanel").style.display = "block"; } console.log("已连接钱包:", userAddress); return true; } catch (e) { console.error(e); // 如果你在 TP 里还报错，可以把下面的 alert 注释掉，避免干扰 // alert("连接失败：" + e.message); return false; } }
